Implementation: GitHub Container Registry (GHCR)

Step 1: VCL Setup - Enable GHCR

# .github/workflows/release.yml
name: Release Docker Image

on:
  push:
    tags:
      - 'v*'

jobs:
  build-and-push:
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write

    steps:
      - uses: actions/checkout@v4

      - name: Login to GHCR
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Build and Push
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: |
            ghcr.io/vcl-trading/ats-health-monitor:${{ github.ref_name }}
            ghcr.io/vcl-trading/ats-health-monitor:latest

Step 2: Create Mega Access Token

# VCL Admin creates a Personal Access Token (PAT) for Mega
# Scope: read:packages only
# No write, no repo access

# Token format: ghp_xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx

Step 3: Mega Setup - Configure Docker Login

# On Mega HK Docker host
echo "ghp_xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx" | docker login ghcr.io -u mega-readonly --password-stdin

# Verify access
docker pull ghcr.io/vcl-trading/ats-health-monitor:latest

Step 4: Mega Update Process

# When VCL announces new version:
cd /opt/ats-health-monitor

# Pull new image
docker pull ghcr.io/vcl-trading/ats-health-monitor:v1.2.3

# Update docker-compose.yml to use new tag
# OR use :latest tag

# Restart
docker compose down
docker compose up -d

# Verify
docker compose logs -f

Alternative: Manual SCP Delivery

If GHCR is not feasible, use manual delivery:

VCL Side (Build & Export)

# Build-and-deliver.ps1
$VERSION = "v1.2.3"
$OUTPUT = "ats-health-monitor-$VERSION.tar"

# Build
docker build -t ats-health-monitor:$VERSION .

# Export
docker save ats-health-monitor:$VERSION -o $OUTPUT

# Compress
gzip $OUTPUT

# SCP to Mega (via WireGuard)
scp "$OUTPUT.gz" megaadmin@10.101.0.40:/opt/ats-health-monitor/

Write-Host "Delivered $OUTPUT.gz to Mega HK"

Mega Side (Import & Deploy)

#!/bin/bash
# update.sh
VERSION=$1

if [ -z "$VERSION" ]; then
    echo "Usage: ./update.sh v1.2.3"
    exit 1
fi

cd /opt/ats-health-monitor

# Decompress
gunzip ats-health-monitor-$VERSION.tar.gz

# Load image
docker load -i ats-health-monitor-$VERSION.tar

# Update docker-compose to use new version
sed -i "s|image: ats-health-monitor:.*|image: ats-health-monitor:$VERSION|" docker-compose.yml

# Restart
docker compose down
docker compose up -d

echo "Updated to $VERSION"
docker compose ps
