# Example: Stale Issue Detection
class StaleIssueMonitor:
    """Detects issues that haven't been updated within SLA"""

    THRESHOLDS = {
        'critical': timedelta(hours=4),
        'high': timedelta(hours=12),
        'medium': timedelta(hours=24),
        'low': timedelta(hours=48)
    }

    def check_issues(self):
        for issue in self.plane.get_open_issues():
            threshold = self.THRESHOLDS.get(issue.priority, timedelta(hours=24))
            if issue.last_updated < datetime.now() - threshold:
                self.trigger_alert(issue)

# Example: Auto-Assignment Logic
TEAM_TIMEZONES = {
    'vancouver': ('America/Vancouver', ['dev-team-a']),
    'chicago': ('America/Chicago', ['ops-team']),
    'hongkong': ('Asia/Hong_Kong', ['trading-ops'])

def assign_to_active_team(issue):
    """Assigns issue to team currently in working hours"""
    now = datetime.now(UTC)

    for location, (tz, teams) in TEAM_TIMEZONES.items():
        local_time = now.astimezone(timezone(tz))
        if is_working_hours(local_time):  # 9am-6pm
            return assign_to_team(issue, teams[0])

    # No team active - assign to next team starting work
    return assign_to_next_shift(issue)
